  // Р Р°Р·СЂРµС€Р°РµРј РґРѕСЃС‚СѓРї Рє СЃС‚Р°С‚РёС‡РµСЃРєРёРј СЂРµСЃСѓСЂСЃР°Рј, API Рё СЃС‚СЂР°РЅРёС†Р°Рј Р°РІС‚РѕСЂРёР·Р°С†РёРё
  if (
    pathname.startsWith("/_next") ||
    pathname.startsWith("/api") ||
    pathname === "/login" ||
    pathname === "/register" ||
    pathname === "/"
  ) {
    return NextResponse.next();
  }

  // РЎС‡РёС‚С‹РІР°РµРј СЃРµСЃСЃРёСЋ РёР· РєСѓРє (СЃРѕР·РґР°РЅРЅСѓСЋ РїСЂРё РІС…РѕРґРµ / СЂРµРіРёСЃС‚СЂР°С†РёРё)
  const session = request.cookies.get("aether-session");

  // Р•СЃР»Рё СЃРµСЃСЃРёРё РЅРµС‚ Рё РїСѓС‚СЊ Р·Р°С‰РёС‰РµРЅ, РїРµСЂРµРЅР°РїСЂР°РІР»СЏРµРј РЅР° Р»РѕРіРёРЅ
  if (!session) {
    if (pathname !== "/login" && pathname !== "/register") {
      const loginUrl = new URL("/login", request.url);
      return NextResponse.redirect(loginUrl);
    }
  } else {
    // Р•СЃР»Рё СЃРµСЃСЃРёСЏ РµСЃС‚СЊ, РЅРµ РґР°РµРј Р·Р°С…РѕРґРёС‚СЊ РЅР° /login, /register Рё / Рё СЃСЂР°Р·Сѓ РїРµСЂРµРЅР°РїСЂР°РІР»СЏРµРј РІРѕ С„СЂРµР№РјРІРѕСЂРє
    if (pathname === "/" || pathname === "/login" || pathname === "/register") {
      const workspaceUrl = new URL("/default-workspace", request.url);
      return NextResponse.redirect(workspaceUrl);
    }
  }

  return NextResponse.next();
}
